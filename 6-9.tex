%% nonumber
\begin{equation*}
  \frac{q}{r_1} + \frac{q'}{r_2}
\end{equation*}

%% nonumber
\begin{equation*}
  \frac{q'}{r_2} = - \frac{q}{r_1}
  \qquad \mathrm{or} \qquad
  \frac{r_2}{r_1} = - \frac{q'}{q}
\end{equation*}

%% Eq. (6.31)
\begin{equation}
  \frac{q'}{q} = - \frac{a}{b}
\end{equation}

%% Eq. (6.32)
\begin{equation}
  q'' = -q' = \frac{a}{b} q
\end{equation}
